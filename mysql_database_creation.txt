create database espdatabase;
use espdatabase;
create table espdata(
Data_id int primary key AUTO_INCREMENT,
Gateway_Time datetime,
Updated_Time datetime,
Count int,
MAC varchar(20),
IP varchar(20),
RSSI int
);
